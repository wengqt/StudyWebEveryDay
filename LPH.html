function make(func,max) {
var last_time = new Date();
return function(){
var cur_time = new Date();
if (cur_time - last_time > max) {
func();
last_time = cur_time;
}
}
}

function doSomething() {
console.log("yess");
}
window.onscroll = make(doSomething,1000);
